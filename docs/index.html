What are "bookmarklets"? They're bookmarks which contain a bit of Javascript code instead of a URL. When you select one from your browser's bookmark bar or bookmark menu, you don't go to a new page, instead you run some code which may affect your current page. They're simpler and easier to install and use than browser extensions.

These particular extensions I made for times when I wanted to write something into a text box on the web, wanted to write some Esperanto, but didn't have the time or inclination to use a proper keyboard. These let you type using the x-system, and then activate a bookmarklet, to help you change the text.

I have two bookmarklets here, one called "CHAPELIGU CHIUJN LA AFEROJN" ("PUT HATS ON ALL THE THINGS"), which will look for every textarea on the page and hat-ify the x-system letters in any of them. The other one, "Kreu chapeligajn butonojn" ("create hat-ifying buttons") will put a button next to each text box. If you click the button, then the x-system letters in that box will receive hats.

On desktop browsers it is usually possible to just drag the links off the page and onto your bookmarks bar; on mobile browsers the process may be more difficult. You can consult a search engine for information on how to do it; here's a good starting place.

function sxangxuCxio() {
    (function() {
        var tas = document.getElementsByTagName('textarea');
        for (var i = 0; i < tas.length; i++) {
            for (var j = 0; j < 12; j++) {
                tas[i].value = tas[i].value.replace(new RegExp('hsgcjuHSGCJU'[j] + '[Xx]', 'g') , ['\u0125', '\u015d', '\u011d', '\u0109', '\u0135','\u016d', '\u0124', '\u015c', '\u011c', '\u0108', '\u0134', '\u016c'][j]);
            }
        }
    })();
}
function aldonuButonojn() {
    (function() {
        var tas = document.getElementsByTagName('textarea');
        var subTxt = function(txt) {
            for (var j = 0; j < 12; j += 1) {
                txt = txt.replace(new RegExp('hsgcjuHSGCJU'[j] + '[Xx]', 'g') , ['\u0125', '\u015d', '\u011d', '\u0109', '\u0135','\u016d', '\u0124', '\u015c', '\u011c', '\u0108', '\u0134', '\u016c'][j]);
            }
            return txt;
        }
        var mkBtn = function(ta) {
            var btn = document.createElement('button')
            btn.innerHTML = '\u0109apeligu';
            btn.onclick = function() { ta.value = subTxt(ta.value); };
            ta.parentNode.insertBefore(btn,tas[i].nextSibling);
        }
        for (var i = 0; i < tas.length; i++) {
            mkBtn(tas[i]);
        }
    })();
}
